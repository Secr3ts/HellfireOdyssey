001import javax.swing.*;
002import java.awt.*;
003import java.awt.event.*;
004import java.net.URL;
005
006/**
007 * This class implements a simple graphical user interface with a text entry
008 * area, a text output area and an optional image.
009 * 
010 * @author Michael Kolling
011 * @version 1.0 (Jan 2003) edited (2023)
012 */
013public class UserInterface implements ActionListener {
014    private GameEngine aEngine;
015    private JFrame aMyFrame;
016    private JTextField aEntryField;
017    private JTextArea aLog;
018    private JLabel aImage;
019    private JPanel aEastPanel;
020    private JPanel aWestPanel;
021
022    private JButton aNorthButton;
023    private JButton aSouthButton;
024    private JButton aEastButton;
025    private JButton aWestButton;
026
027    private JButton aTakeButton;
028    private JButton aDropButton;
029    private JButton aPrayButton;
030    private JButton aLookButton;
031
032    /**
033     * Construct a UserInterface. As a parameter, a Game Engine
034     * (an object processing and executing the game commands) is
035     * needed.
036     * 
037     * @param pGameEngine The GameEngine object implementing the game logic.
038     */
039    public UserInterface(final GameEngine pGameEngine) {
040        this.aEngine = pGameEngine;
041        this.createGUI();
042    };
043
044    /**
045     * Print out some text into the text area.
046     */
047    public void print(final String pText) {
048        this.aLog.append(pText);
049        this.aLog.setCaretPosition(this.aLog.getDocument().getLength());
050    } // print(.)
051
052    /**
053     * Print out some text into the text area, followed by a line break.
054     */
055    public void println(final String pText) {
056        this.print(pText + "\n");
057    } // println(.)
058
059    /**
060     * Show an image file in the interface.
061     */
062    public void showImage(final String pImageName) {
063        String vImagePath = "" + pImageName; // to change the directory
064        URL vImageURL = this.getClass().getClassLoader().getResource(vImagePath);
065        if (vImageURL == null)
066            System.out.println("Image not found : " + vImagePath);
067        else {
068            ImageIcon vIcon = new ImageIcon(vImageURL);
069            Image image = vIcon.getImage();
070            image = image.getScaledInstance(600, 600, Image.SCALE_SMOOTH);
071            vIcon = new ImageIcon(image);
072            this.aImage.setIcon(vIcon);
073            this.aMyFrame.pack();
074        }
075    } // showImage(.)
076
077    /**
078     * Enable or disable input in the input field.
079     */
080    public void enable(final boolean pOnOff) {
081        this.aEntryField.setEditable(pOnOff); // enable/disable
082        if (!pOnOff) { // disable
083            this.aEntryField.getCaret().setBlinkRate(0); // cursor won't blink
084            this.aEntryField.removeActionListener(this); // won't react to entry
085        }
086    } // enable(.)
087
088    /**
089     * Set up graphical user interface.
090     */
091    private void createGUI() {
092        // The current aMyFrame customization
093        this.aMyFrame = new JFrame("Hellfire Odyssey"); // change the title
094        this.aEntryField = new JTextField(34);
095
096        // icons
097        SystemTray tray = SystemTray.getSystemTray();
098        Image image = Toolkit.getDefaultToolkit().getImage("Images/favicon.jpeg");
099        TrayIcon trayIcon = new TrayIcon(image, "Hellfire Odyssey");
100        trayIcon.setImageAutoSize(true);
101        try {
102            tray.add(trayIcon);
103        } catch (AWTException e) {
104            System.err.println(e);
105        }
106
107        this.aMyFrame.setIconImage(Toolkit.getDefaultToolkit().getImage("Images/favicon.jpeg"));
108
109        this.aLog = new JTextArea();
110        this.aLog.setEditable(false);
111        JScrollPane vListScroller = new JScrollPane(this.aLog);
112        vListScroller.setPreferredSize(new Dimension(200, 200));
113        vListScroller.setMinimumSize(new Dimension(100, 100));
114
115        JPanel vPanel = new JPanel();
116        this.aImage = new JLabel();
117
118        vPanel.setLayout(new BorderLayout()); // ==> only five places
119        vPanel.add(this.aImage, BorderLayout.NORTH);
120        vPanel.add(vListScroller, BorderLayout.CENTER);
121        vPanel.add(this.aEntryField, BorderLayout.SOUTH);
122
123        this.aMyFrame.getContentPane().add(vPanel, BorderLayout.CENTER);
124
125        // add 2 other panes, East and West.
126        this.aEastPanel = new JPanel();
127        this.aWestPanel = new JPanel();
128
129        // set grid layout, 4x1 for both
130        this.aEastPanel.setLayout(new GridLayout(4, 1));
131        this.aWestPanel.setLayout(new GridLayout(4, 1));
132
133        // initialize buttons and applies them to their respective panels. East will be
134        // for direction, west for actions. (temp)
135
136        // North Direction
137        this.aNorthButton = new JButton("North", null);
138        this.aNorthButton.addActionListener(this);
139        this.aNorthButton.setBackground(Color.black);
140        this.aEastPanel.add(this.aNorthButton);
141        this.aNorthButton.setForeground(Color.white);
142
143        // South Direction
144        this.aSouthButton = new JButton("South", null);
145        this.aSouthButton.addActionListener(this);
146        this.aSouthButton.setBackground(Color.black);
147        this.aEastPanel.add(this.aSouthButton);
148        this.aSouthButton.setForeground(Color.white);
149
150        // East Direction
151        this.aEastButton = new JButton("East", null);
152        this.aEastButton.addActionListener(this);
153        this.aEastButton.setBackground(Color.black);
154        this.aEastPanel.add(this.aEastButton);
155        this.aEastButton.setForeground(Color.white);
156
157        // West Direction
158        this.aWestButton = new JButton("West", null);
159        this.aWestButton.addActionListener(this);
160        this.aWestButton.setBackground(Color.black);
161        this.aEastPanel.add(this.aWestButton);
162        this.aWestButton.setForeground(Color.white);
163
164        // Look Action
165        this.aLookButton = new JButton("Look", null);
166        this.aLookButton.addActionListener(this);
167        this.aLookButton.setBackground(Color.ORANGE);
168        this.aWestPanel.add(this.aLookButton);
169        this.aWestButton.setForeground(Color.white);
170
171        // Take Action
172        this.aTakeButton = new JButton("Back", null);
173        this.aTakeButton.addActionListener(this);
174        this.aTakeButton.setBackground(Color.ORANGE);
175        this.aWestPanel.add(this.aTakeButton);
176        this.aTakeButton.setForeground(Color.white);
177
178        // Drop Action
179        this.aDropButton = new JButton("null", null);
180        this.aDropButton.addActionListener(this);
181        this.aDropButton.setBackground(Color.ORANGE);
182        this.aWestPanel.add(this.aDropButton);
183        this.aDropButton.setForeground(Color.white);
184
185        // Pray Action
186        this.aPrayButton = new JButton("Pray", null);
187        this.aPrayButton.addActionListener(this);
188        this.aPrayButton.setBackground(Color.ORANGE);
189        this.aWestPanel.add(this.aPrayButton);
190        this.aPrayButton.setForeground(Color.white);
191
192        vPanel.add(this.aWestPanel, BorderLayout.WEST);
193        vPanel.add(this.aEastPanel, BorderLayout.EAST);
194        // add some event listeners to some components
195        this.aEntryField.addActionListener(this);
196
197        // to end program when window is closed
198        this.aMyFrame.addWindowListener(new WindowAdapter() {
199            public void windowClosing(WindowEvent e) {
200                System.exit(0);
201            }
202        });
203
204        this.aMyFrame.pack();
205        this.aMyFrame.setVisible(true);
206        this.aEntryField.requestFocus();
207    } // createGUI()
208
209    /**
210     * Actionlistener interface for entry textfield.
211     */
212    public void actionPerformed(final ActionEvent pE) {
213        if (pE.getSource() == this.aEntryField)
214            this.processCommand();
215        else if (pE.getSource() == this.aNorthButton)
216            this.aEngine.interpretCommand("go north");
217        else if (pE.getSource() == this.aSouthButton)
218            this.aEngine.interpretCommand("go south");
219        else if (pE.getSource() == this.aEastButton)
220            this.aEngine.interpretCommand("go east");
221        else if (pE.getSource() == this.aWestButton)
222            this.aEngine.interpretCommand("go west");
223        else if (pE.getSource() == this.aLookButton)
224            this.aEngine.interpretCommand("look");
225        else if (pE.getSource() == this.aTakeButton)
226            this.aEngine.interpretCommand("back");
227        else if (pE.getSource() == this.aDropButton)
228            this.aEngine.interpretCommand("");
229        else if (pE.getSource() == this.aPrayButton)
230            this.aEngine.interpretCommand("pray");
231        else
232            System.out.println("Unexpected action event source.");
233    } // actionPerformed(.)
234
235    /**
236     * A command has been entered. Read the command and do whatever is
237     * necessary to process it.
238     */
239    private void processCommand() {
240        String vInput = this.aEntryField.getText();
241        this.aEntryField.setText("");
242
243        this.aEngine.interpretCommand(vInput);
244    } // processCommand()
245} // UserInterface
